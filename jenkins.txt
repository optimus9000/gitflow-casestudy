pipeline {
    agent any
    environment {
        DOCKER_IMAGE = 'my-ubuntu-apache'
    }
    stages {
        stage('Build') {
            steps {
                script {
                    if (env.BRANCH_NAME == 'master') {
                        sh 'docker build -t $DOCKER_IMAGE .'
                        sh 'docker run -d -p 82:80 -v $WORKSPACE:/var/www/html $DOCKER_IMAGE'
                    } else if (env.BRANCH_NAME == 'develop') {
                        sh 'docker build -t $DOCKER_IMAGE .'
                    }
                }
            }
        }
    }
    post {
        always {
            cleanWs()
        }
    }
}
